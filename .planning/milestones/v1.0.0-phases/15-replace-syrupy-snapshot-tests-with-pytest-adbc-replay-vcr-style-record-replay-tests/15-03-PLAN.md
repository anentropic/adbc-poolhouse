Create all cassette directories and their triplet files using a Python script. Run this with `uv run python`:

```python
#!/usr/bin/env python
"""Generate synthetic cassette files for pytest-adbc-replay CI replay."""
from pathlib import Path
import pyarrow as pa
import pyarrow.ipc as ipc

CASSETTE_DIR = Path("tests/cassettes")

def make_arrow_ipc(table: pa.Table) -> bytes:
    """Serialize a pyarrow Table to Arrow IPC file format bytes."""
    sink = pa.BufferOutputStream()
    writer = ipc.new_file(sink, table.schema)
    writer.write_table(table)
    writer.close()
    return sink.getvalue().to_pybytes()

def write_cassette(
    cassette_name: str,
    driver_module: str,
    sql: str,
    table: pa.Table,
) -> None:
    """Write a .sql + .arrow + .json triplet for one query."""
    d = CASSETTE_DIR / cassette_name / driver_module
    d.mkdir(parents=True, exist_ok=True)
    (d / "000.sql").write_text(sql, encoding="utf-8")
    (d / "000.arrow").write_bytes(make_arrow_ipc(table))
    (d / "000.json").write_text("null", encoding="utf-8")

# --- Snowflake: SELECT 1
# Column name from Snowflake for SELECT 1 is "1" (unaliased integer literal)
snowflake_health_table = pa.table({"1": pa.array([1], type=pa.int64())})
write_cassette(
    "snowflake_health",
    "adbc_driver_snowflake.dbapi",
    "SELECT 1",
    snowflake_health_table,
)

# --- Snowflake: SELECT 1 AS n, 'hello' AS s
snowflake_arrow_table = pa.table({
    "n": pa.array([1], type=pa.int64()),
    "s": pa.array(["hello"], type=pa.string()),
})
write_cassette(
    "snowflake_arrow_round_trip",
    "adbc_driver_snowflake.dbapi",
    "SELECT 1 AS n, 'hello' AS s",
    snowflake_arrow_table,
)

# --- Databricks: SELECT 1
# Databricks also returns unaliased SELECT 1 column as "1"
databricks_health_table = pa.table({"1": pa.array([1], type=pa.int64())})
write_cassette(
    "databricks_health",
    "adbc_driver_manager.dbapi",
    "SELECT 1",
    databricks_health_table,
)

# --- Databricks: SELECT 1 AS n, 'hello' AS s
databricks_arrow_table = pa.table({
    "n": pa.array([1], type=pa.int64()),
    "s": pa.array(["hello"], type=pa.string()),
})
write_cassette(
    "databricks_arrow_round_trip",
    "adbc_driver_manager.dbapi",
    "SELECT 1 AS n, 'hello' AS s",
    databricks_arrow_table,
)

print("Cassette files written:")
for p in sorted(CASSETTE_DIR.rglob("*")):
    if p.is_file():
        print(f"  {p}  ({p.stat().st_size} bytes)")
```

**IMPORTANT NOTE on SQL normalization:** `pytest-adbc-replay` uses `sqlglot` to normalize SQL before matching cassettes. The `000.sql` files should contain the exact SQL string the test will execute. If the plugin normalizes SQL (e.g., uppercasing, adding semicolons), the stored SQL may not need to match exactly — but starting with the exact string the test uses is safest.

**IMPORTANT NOTE on column names:** The actual column name returned by Snowflake for `SELECT 1` (unaliased) may differ from `"1"`. It could be `"1"`, `"_col0"`, `"?COLUMN?"`, etc. — this depends on the driver. Since we're creating synthetic cassettes (not recording from live), the column name in the Arrow schema must match what the test assertions check. The tests check `row[0] == 1` (positional, not by name) and `table.num_rows == 1` — so the exact column name doesn't matter for test assertions. Use `"1"` as a reasonable placeholder.

**IMPORTANT NOTE on Arrow IPC version:** If the plugin expects a specific IPC format version or schema metadata, the synthetic cassettes may not match on first replay attempt. If tests fail with a cassette format error, try using `ipc.new_stream()` instead of `ipc.new_file()`, or check the plugin's cassette reading code for format requirements.

**Fallback if Arrow IPC format doesn't match:** The executor should check the pytest-adbc-replay source or documentation for the exact Arrow IPC format (file vs stream). Try both `ipc.new_file()` and `ipc.new_stream()` and use whichever makes the tests pass.

```python
"""Snowflake integration tests: cassette-based, CI-safe (TEST-03 successor)."""

from __future__ import annotations

import adbc_driver_snowflake.dbapi  # intercepted by adbc_auto_patch in replay mode
import pytest


@pytest.mark.snowflake
@pytest.mark.adbc_cassette("snowflake_health")
def test_connection_health() -> None:
    """Pool path works: connect + SELECT 1 + checkin.

    In CI: replayed from tests/cassettes/snowflake_health/ (no credentials required).
    To record: pytest --adbc-record=once -m snowflake (requires SNOWFLAKE_ACCOUNT set).
    """
    conn = adbc_driver_snowflake.dbapi.connect()
    cur = conn.cursor()
    cur.execute("SELECT 1")
    row = cur.fetchone()
    assert row is not None
    assert row[0] == 1
    cur.close()
    conn.close()


@pytest.mark.snowflake
@pytest.mark.adbc_cassette("snowflake_arrow_round_trip")
def test_arrow_round_trip() -> None:
    """Arrow schema + rows round-trip correctly; cassette enforces stable output.

    In CI: replayed from tests/cassettes/snowflake_arrow_round_trip/ (no credentials required).
    To record: pytest --adbc-record=once -m snowflake (requires SNOWFLAKE_ACCOUNT set).
    """
    conn = adbc_driver_snowflake.dbapi.connect()
    cur = conn.cursor()
    cur.execute("SELECT 1 AS n, 'hello' AS s")
    table = cur.fetch_arrow_table()
    cur.close()
    conn.close()
    assert table is not None
    assert table.num_rows == 1
```

```python
import pytest

@pytest.mark.adbc_cassette("databricks_health")

    assert row[0] == 1
    cur.close()
    conn.close()
```

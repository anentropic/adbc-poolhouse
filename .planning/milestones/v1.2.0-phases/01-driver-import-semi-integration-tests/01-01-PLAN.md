    ```python
    class TestDuckDBImports:
        """Semi-integration test: real driver import, mocked connection."""

        def test_create_pool_wiring(self) -> None:
            """Import real DuckDB driver, mock connection, assert correct kwargs."""
            from unittest.mock import MagicMock, patch

            from adbc_poolhouse import DuckDBConfig, create_pool

            config = DuckDBConfig()  # :memory: by default
            mock_conn = MagicMock()
            mock_conn.adbc_clone = MagicMock(return_value=MagicMock())

            with patch(
                "adbc_poolhouse._pool_factory.create_adbc_connection",
                return_value=mock_conn,
            ) as mock_factory:
                pool = create_pool(config)
                pool.dispose()

            mock_factory.assert_called_once()
            call_args = mock_factory.call_args
            # DuckDB: assert entrypoint='duckdb_adbc_init'
            assert call_args.kwargs.get("entrypoint") == "duckdb_adbc_init"
    ```

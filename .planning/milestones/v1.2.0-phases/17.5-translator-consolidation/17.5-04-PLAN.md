    ```python
    from urllib.parse import quote
    
    def to_adbc_kwargs(self) -> dict[str, str]:
        """Convert PostgreSQLConfig to ADBC driver kwargs.
        
        Supports two modes:
        - URI mode: {uri: ...} when uri is set
        - Decomposed mode: libpq URI from individual fields
        - Empty mode: {} (let libpq resolve from env)
        """
        if self.uri is not None:
            return {"uri": self.uri}
        
        # Decomposed mode - check if any fields are set
        has_fields = any([
            self.host, self.user, self.password, self.database, self.sslmode
        ])
        if not has_fields:
            return {}
        
        # Build URI
        uri = "postgresql://"
        if self.user is not None:
            uri += quote(self.user, safe="")
            if self.password is not None:
                uri += ":" + quote(self.password.get_secret_value(), safe="")
            uri += "@"
        if self.host is not None:
            uri += self.host
        if self.port is not None:
            uri += f":{self.port}"
        if self.database is not None:
            uri += f"/{self.database}"
        if self.sslmode is not None:
            uri += f"?sslmode={self.sslmode}"
        
        return {"uri": uri}
    ```

    ```python
    from urllib.parse import quote
    
    def to_adbc_kwargs(self) -> dict[str, str]:
        """Convert RedshiftConfig to ADBC driver kwargs.
        
        Supports two modes:
        - URI mode: {uri: ...} plus IAM fields
        - Decomposed mode: redshift:// URI plus IAM fields
        """
        result: dict[str, str] = {}
        
        # URI handling
        if self.uri is not None:
            result["uri"] = self.uri
        else:
            # Build URI from fields
            uri = "redshift://"
            if self.user is not None:
                uri += quote(self.user, safe="")
                if self.password is not None:
                    uri += ":" + quote(self.password.get_secret_value(), safe="")
                uri += "@"
            if self.host is not None:
                uri += self.host
            if self.database is not None:
                uri += f"/{self.database}"
            result["uri"] = uri
        
        # IAM fields (added as separate kwargs, not in URI)
        if self.cluster_type is not None:
            result["redshift.cluster_type"] = self.cluster_type
        if self.cluster_identifier is not None:
            result["redshift.cluster_identifier"] = self.cluster_identifier
        if self.aws_region is not None:
            result["aws_region"] = self.aws_region
        if self.aws_secret_access_key is not None:
            result["aws_secret_access_key"] = self.aws_secret_access_key.get_secret_value()
        
        return result
    ```

    Implement `to_adbc_kwargs()` in TrinoConfig:
    
    Read _trino_translator.py first. Key patterns:

    - Decomposed mode: host, catalog, port, user, password (SecretStr), ssl, ssl_verify
    - ssl defaults to True, ssl_verify defaults to True
    
    Implementation:
    ```python
    @field_serializer("password", mode="plain")
    def serialize_secret(self, value: SecretStr | None) -> str | None:
        if value is None:
            return None
        return value.get_secret_value()
    
    @field_serializer("ssl", "ssl_verify", mode="plain")
    def serialize_bool(self, value: bool) -> str:
        return str(value).lower()
    
    def to_adbc_kwargs(self) -> dict[str, str]:
        """Convert TrinoConfig to ADBC driver kwargs.
        
        Supports two modes:
        - URI mode: {uri: ...} when uri is set
        - Decomposed mode: individual fields with boolean defaults
        """
        if self.uri is not None:
            return {"uri": self.uri}
        
        # Decomposed mode
        result: dict[str, str] = {}
        if self.host is not None:
            result["host"] = self.host
        if self.catalog is not None:
            result["catalog"] = self.catalog
        if self.port is not None:
            result["port"] = str(self.port)
        if self.user is not None:
            result["user"] = self.user
        if self.password is not None:
            result["password"] = self.password.get_secret_value()
        result["ssl"] = str(self.ssl).lower()
        result["ssl_verify"] = str(self.ssl_verify).lower()
        return result
    ```
    
    Note: ADBC key names for Trino are LOW confidence (from translator comment).
